DROP TRIGGER AUDIT_STAG_TEST.TRG_VSL_UPR_CASE_VESSEL_REFRESH;

CREATE OR REPLACE TRIGGER AUDIT_STAG_TEST.trg_vsl_upr_case_vessel_refresh
    -- this trigger is to make vessel type of ma_vessel_yatch table as upper case
    BEFORE INSERT OR UPDATE ON AUDIT_STAG_TEST.VESSEL_REFRESH     FOR EACH ROW
BEGIN

      :NEW.VESSELTYPE := UPPER(:NEW.VESSELTYPE);
    END;
/
