DROP TRIGGER AUDIT_STAG_TEST.COMPANY_IUD_TRG;

CREATE OR REPLACE TRIGGER AUDIT_STAG_TEST.company_iud_trg 
after insert or delete or update 
ON AUDIT_STAG_TEST.MA_COMPANY 
begin
 if inserting then 
 update master_table_update set table_updation=0 where table_name='Company';
 end if;
 if deleting then
 update master_table_update set table_updation=0 where table_name='Company';
 end if;

 if updating then 
 update master_table_update set table_updation=0 where table_name='Company';
 end if;
 end ;
/
