DROP TRIGGER AUDIT_STAG_TEST.UPDATE_FINDING_AUDIT_DATE;

CREATE OR REPLACE TRIGGER AUDIT_STAG_TEST.UPDATE_FINDING_AUDIT_DATE
   AFTER UPDATE OF AUDIT_DATE
   ON AUDIT_STAG_TEST.AUDIT_DETAILS    FOR EACH ROW
DECLARE
 v_Mgr varchar2(200);
 
   --added because of getting mutating exception
   PRAGMA AUTONOMOUS_TRANSACTION;
   
BEGIN
  
  UPDATE AUDIT_FINDINGS SET AUDIT_DATE=:NEW.AUDIT_DATE WHERE AUDIT_SEQ_NO=:OLD.AUDIT_SEQ_NO ;
  COMMIT;
  
END;
/
