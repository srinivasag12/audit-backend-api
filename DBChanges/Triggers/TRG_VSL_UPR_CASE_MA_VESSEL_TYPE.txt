DROP TRIGGER AUDIT_STAG_TEST.TRG_VSL_UPR_CASE_MA_VESSEL_TYPE;

CREATE OR REPLACE TRIGGER AUDIT_STAG_TEST.trg_vsl_upr_case_ma_vessel_type
    -- this trigger is to make vessel type of ma_vessel_type table as upper case
    BEFORE INSERT OR UPDATE ON AUDIT_STAG_TEST.MA_VESSEL_TYPE     FOR EACH ROW
BEGIN

      :NEW.VESSEL_TYPE_NAME := UPPER(:NEW.VESSEL_TYPE_NAME);
    END;
/
