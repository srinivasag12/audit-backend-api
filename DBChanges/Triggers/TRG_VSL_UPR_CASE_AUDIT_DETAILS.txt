DROP TRIGGER AUDIT_STAG_TEST.TRG_VSL_UPR_CASE_AUDIT_DETAILS;

CREATE OR REPLACE TRIGGER AUDIT_STAG_TEST.trg_vsl_upr_case_Audit_details
    -- this trigger is to make vessel type of audit_details table as upper case
    BEFORE INSERT OR UPDATE ON AUDIT_STAG_TEST.AUDIT_DETAILS     FOR EACH ROW
BEGIN

      :NEW.VESSEL_TYPE := UPPER(:NEW.VESSEL_TYPE);
    END;
/
