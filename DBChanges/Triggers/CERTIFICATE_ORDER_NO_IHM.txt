DROP TRIGGER AUDIT_STAG_TEST.CERTIFICATE_ORDER_NO_IHM;

CREATE OR REPLACE TRIGGER AUDIT_STAG_TEST."CERTIFICATE_ORDER_NO_IHM" 
BEFORE INSERT
ON AUDIT_STAG_TEST.CERTIFICATE_IHM_DETAIL 
REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
BEGIN
  SELECT CERTIFICATE_ORDER_NO.NEXTVAL
  INTO   :new.CERT_ORDER_NO
  FROM DUAL ;
 
END;
/
