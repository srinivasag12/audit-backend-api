CREATE OR REPLACE function AUDIT_STAG_TEST.validate_date_string (
     p_date_string varchar2
)
return varchar2
is
l_date date;
l_date_string varchar2(4000);
begin
    l_date_string := p_date_string;
    begin
        l_date := to_date (p_date_string, 'DD-MON-RRRR');
    exception
        when others then l_date_string := '000000'; 
    end;
    return l_date_string;
end;
/
