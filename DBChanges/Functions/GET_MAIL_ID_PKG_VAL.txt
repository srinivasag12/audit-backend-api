CREATE OR REPLACE FUNCTION AUDIT_STAG_TEST.get_mail_id_pkg_val (p_pkg_var_name in varchar2)
  RETURN VARCHAR2
AS
  v_from_id   VARCHAR2 (100); 
  v_vesDocs_id VARCHAR2 (100);     
        
BEGIN

  v_from_id := audit_details_pkg.v_from;
  v_vesDocs_id := audit_details_pkg.V_VESDOC_DEPT;

  RETURN v_from_id || ',' || v_vesDocs_id;
END get_mail_id_pkg_val;
/
