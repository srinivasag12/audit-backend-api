CREATE OR REPLACE procedure AUDIT_STAG_TEST.new_emp_create_mail(employeeid varchar2,employeename varchar2)
as

 v_to varchar2(60):='parashuram.jumanal@bsolsystem.com';
 v_sub varchar(200);
 v_body varchar(200);

BEGIN

     FROM audit_email_dtls
    WHERE email_seq = 402;
      select replace(v_body,'employee name',employeename) into v_body from audit_email_dtls where email_seq=402;

 end;
